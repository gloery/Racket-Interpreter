(define not
  (lambda (bool)
    (if bool #f #t)))
(letrec ((is-even? (lambda (n)
                       (or (= n 0)
                           (is-odd? (- n 1)))))
           (is-odd? (lambda (n)
                      (and (not (= n 0))
                           (is-even? (- n 1))))))
    (is-odd? 10))
